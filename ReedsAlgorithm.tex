\section{–ê–ª–≥–æ—Ä–∏—Ç–º –†–∏–¥–∞: –æ–¥–Ω–æ–∑–Ω–∞—á–Ω–æ–µ –¥–µ–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–∏–µ –≤ —Ä–∞–º–∫–∞—Ö –∫–æ—Ä—Ä–µ–∫—Ç–∏—Ä—É—é—â–µ–π —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç–∏}

–í —ç—Ç–æ–º —Ä–∞–∑–¥–µ–ª–µ –æ–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –∞–ª–≥–æ—Ä–∏—Ç–º –†–∏–¥–∞ –¥–ª—è $\RM(r, m)$. –û–Ω –∏—Å–ø—Ä–∞–≤–ª—è–µ—Ç –ª—é–±—ã–µ –æ—à–∏–±–∫–∏, –≤–µ—Å –∫–æ—Ç–æ—Ä—ã—Ö –Ω–µ –ø—Ä–µ–≤—ã—à–∞–µ—Ç $2^{m-r-1}$, –ø–æ–ª–æ–≤–∏–Ω—É –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–≥–æ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –∫–æ–¥–∞.

–î–ª—è –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–∞ $A ‚äÜ \{1‚Ä¶m\}$ –æ–ø—Ä–µ–¥–µ–ª–∏–º $\overline{A} = \{1‚Ä¶m\} ‚àñ A$. –û–±–æ–∑–Ω–∞—á–∏–º –º–æ–Ω–æ–º $x_A = \prod_{i‚ààA} x_i$, –≥–¥–µ $x_i$ ‚Äî –∞—Ä–≥—É–º–µ–Ω—Ç—ã –±—É–ª–µ–≤–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ \gray{[–Ω–∞–ø—Ä., $x_{\{1,2\}} = x_1x_2$]}. –¢–∞–∫–∂–µ –±—É–¥–µ–º –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å $V_A := \{z ‚àà ùîΩ_2^m : z_i = 0 \,‚àÄi \not\in A \}$ –¥–ª—è –æ–±–æ–∑–Ω–∞—á–µ–Ω–∏—è –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞ $ùîΩ_2^m$ —Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç–∏ $|A|$, —Ç.–µ. $V_A$ —ç—Ç–æ –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–æ, –≤ –∫–æ—Ç–æ—Ä–æ–º –¥–ª—è –≤—Å–µ—Ö –≤–µ–∫—Ç–æ—Ä–æ–≤ $z$ –∑–∞—Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω—ã –±–∏—Ç—ã $z_i = 0$ –ø—Ä–∏ $i \not\in A$.
–î–ª—è –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞ $V_A$ (–º–æ—â–Ω–æ—Å—Ç–∏ $2^{|A|}$ –≤ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ $ùîΩ_2^m$) —Å—É—â–µ—Å—Ç–≤—É–µ—Ç $2^{m - |A|}$ —Å–º–µ–∂–Ω—ã—Ö –∫–ª–∞—Å—Å–∞ –≤–∏–¥–∞ $V_A + b := \{z + b \mid z ‚àà V_A\}$, –≥–¥–µ —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–æ $b ‚àà F_2^m$.
–¢–æ–≥–¥–∞ –¥–ª—è –ª—é–±–æ–≥–æ $A ‚äÜ \{1,‚Ä¶,m\}$ –∏ $b ‚àà ùîΩ_2^m$ –º—ã –∏–º–µ–µ–º \[\sum_{z‚àà(V_A + b)} \Eval_z(x_A) = 1,\] –∞ –¥–ª—è –ª—é–±—ã—Ö $A \not‚äÜ B$, \[\sum_{z‚àà(V_A + b)} \Eval_z(x_B) = 0\]
–≠—Ç–∏ –¥–≤–µ —Å—É–º–º—ã –Ω–∞–¥ $ùîΩ_2$ \gray{[—Ç.–µ. 1 + 1 + 1 = 1]}.
–ü–µ—Ä–≤–∞—è —Å—É–º–º–∞ –≤—ã—Ç–µ–∫–∞–µ—Ç –∏–∑ —Ç–æ–≥–æ, —á—Ç–æ $\Eval_z(x_A) = 1$ –µ—Å–ª–∏ –∏ —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ $z_i = 1\, ‚àÄi‚ààA$, –ø—Ä–∏—á—ë–º —Å—É—â–µ—Å—Ç–≤—É–µ—Ç —Ç–æ–ª—å–∫–æ –æ–¥–∏–Ω —Ç–∞–∫–æ–π $z ‚àà (V_A + b)$ \gray{[–ø–æ—è—Å–Ω–µ–Ω–∏–µ: $\Eval_z(x_A) = \Eval_z(x_{A_1}x_{A_2}‚Ä¶x_{A_k})$, —á—Ç–æ –¥–µ–π—Å—Ç–≤–∏—Ç–µ–ª—å–Ω–æ –≤–µ—Ä–Ω–æ —Ç–æ–ª—å–∫–æ –ø—Ä–∏ –≤—Å–µ—Ö –µ–¥–∏–Ω–∏—Ü–∞—Ö, –∏ –≤ $(V_A + b)$ —Ç–∞–∫–æ–π –≤–µ–∫—Ç–æ—Ä –±—É–¥–µ—Ç –æ–¥–∏–Ω –∏–∑ —Ç–æ–≥–æ, –∫–∞–∫ —É—Å—Ç—Ä–æ–µ–Ω–æ $V_A$]}.
–î–ª—è –¥–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–∞ –≤—Ç–æ—Ä–æ–π —Å—É–º–º—ã, –Ω—É–∂–Ω–æ –∑–∞–º–µ—Ç–∏—Ç—å, —á—Ç–æ –ø–æ—Å–∫–æ–ª—å–∫—É $A\not‚äÜB$, —Ç–æ $‚àÉi ‚àà A‚àñB$, –∞ –∑–Ω–∞—á–∏—Ç –±–∏—Ç $z_i$ –Ω–µ –≤–ª–∏—è–µ—Ç –Ω–∞ –∑–Ω–∞—á–µ–Ω–∏–µ $\Eval_z(x_B)$. –û—Ç—Å—é–¥–∞, $\Eval_{z,z_i = 0}(x_B) = \Eval_{z,z_i = 1}(x_B)$, –∞ –∑–Ω–∞—á–∏—Ç –≤—Å–µ –µ–¥–∏–Ω–∏—á–∫–∏ –≤ —ç—Ç–æ–π —Å—É–º–º–µ –≤–∑–∞–∏–º–æ—É–Ω–∏—á—Ç–æ–∂–∞—Ç—Å—è.

–ü—Ä–µ–¥–ø–æ–ª–æ–∂–∏–º, —á—Ç–æ –±–∏—Ç–æ–≤—ã–π –≤–µ–∫—Ç–æ—Ä $y = \left(y_z \mid z ‚àà ùîΩ_2^m\right)$ ‚Äî –∑–∞—à—É–º–ª—ë–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è –∫–æ–¥–æ–≤–æ–≥–æ —Å–ª–æ–≤–∞ $\Eval(f) ‚àà \RM(r, m)$, —Ç–∞–∫–æ–≥–æ —á—Ç–æ $y$ –∏ $\Eval(f)$ –æ—Ç–ª–∏—á–∞—é—Ç—Å—è –Ω–µ –±–æ–ª–µ–µ —á–µ–º –≤ $2^{m - r - 1}$ –ø–æ–∑–∏—Ü–∏—è—Ö. –ê–ª–≥–æ—Ä–∏—Ç–º –†–∏–¥–∞ –ø–æ–∑–≤–æ–ª—è–µ—Ç –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–∏—Ç—å –∏—Å—Ö–æ–¥–Ω–æ–µ –∫–æ–¥–æ–≤–æ–µ —Å–ª–æ–≤–æ –∏–∑ $y$, –∏–∑–≤–ª–µ–∫–∞—è –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã –ø–æ–ª–∏–Ω–æ–º–∞ $f$. –ü–æ—Å–∫–æ–ª—å–∫—É $\deg f ‚â§ r$, –º—ã –≤—Å–µ–≥–¥–∞ —ç—Ç–æ –º–æ–∂–µ–º –∑–∞–ø–∏—Å–∞—Ç—å $f = \sum_{A‚äÜ\{1,‚Ä¶,m\},|A|‚â§r} u_Ax_A$, –≥–¥–µ $u_A$ ‚Äî –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏—Ö –º–æ–Ω–æ–º–æ–≤. –ê–ª–≥–æ—Ä–∏—Ç–º –†–∏–¥–∞ —Å–Ω–∞—á–∞–ª–∞ –∏–∑–≤–ª–µ–∫–∞–µ—Ç –≤—Å–µ –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã –ø—Ä–∏ –º–æ–Ω–º–∞—Ö —Å—Ç–µ–ø–µ–Ω–∏ $r$, –∑–∞—Ç–µ–º –ø—Ä–∏ —Å—Ç–µ–ø–µ–Ω–∏ $r-1$, –∏ —Ç–∞–∫ –¥–∞–ª–µ–µ –ø–æ–∫–∞ –Ω–µ –Ω–∞–π–¥—ë—Ç –∏—Ö –≤—Å–µ.

–ß—Ç–æ–±—ã –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–∏—Ç—å –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç $u_A$ –ø—Ä–∏ $|A| = r$ \gray{[–ø—Ä–∏ –º–æ–Ω–æ–º–µ —Å—Ç–µ–ø–µ–Ω–∏ $r$]}, –∞–ª–≥–æ—Ä–∏—Ç–º –†–∏–¥–∞ –≤—ã—á–∏—Å–ª—è–µ—Ç —Å—É–º–º—É $\sum_{z‚àà(V_A + b)} y_z$ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∏–∑ $2^{m - r}$ —Å–º–µ–∂–Ω—ã—Ö –∫–ª–∞—Å—Å–æ–≤ –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞ $V_A$, –∞ –∑–∞—Ç–µ–º –≤—ã–±–∏—Ä–∞–µ—Ç –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –±–æ–ª—å—à–∏–Ω—Å—Ç–≤–æ–º –≥–æ–ª–æ—Å–æ–≤ —Å—Ä–µ–¥–∏ —ç—Ç–∏—Ö $2^{m - r}$ —Å—É–º–º. –ï—Å–ª–∏ —Ç–∞–º –±–æ–ª—å—à–µ –µ–¥–∏–Ω–∏—Ü, —á–µ–º –Ω—É–ª–µ–π, —Ç–æ –≤–æ—Å—Å—Ç–∞–Ω–∞–ª–∏–≤–∞–µ–º $u_A = 1$, –∏–Ω–∞—á–µ $u_A = 0$. –ó–∞–º–µ—Ç–∏–º, —á—Ç–æ –µ—Å–ª–∏ $y = \Eval(f)$, —Ç.–µ. –æ—à–∏–±–∫–∏ –Ω–µ—Ç, —Ç–æ:
\[
    \sum_{s‚àà(V_A + b)} y_z
    = \sum_{s ‚àà (V_A + b)} \Eval_z\left(\sum_{\substack{B ‚äÜ \{1,‚Ä¶,m\}\\|B|‚â§r}} u_Bx_B \right)
    = \sum_{\substack{B ‚äÜ \{1,‚Ä¶,m\}\\|B|‚â§r}} u_B \sum_{s ‚àà (V_A + b)} \Eval_z(x_B).
\]

–ò–∑ –ø–æ–ª—É—á–µ–Ω–Ω—ã—Ö —Ä–∞–Ω–µ–µ —Ä–∞–≤–µ–Ω—Å—Ç–≤ –∏ –ø—Ä–∏ —É—Å–ª–æ–≤–∏–∏, —á—Ç–æ $B ‚äÜ \{1,‚Ä¶,m\}$ –∏ $|B| ‚â§ r = |A|$, –ø–æ–ª—É—á–∞–µ–º $\sum_{z‚àà(V_A + b)}\Eval_z(x_B) = 1$ —Ç–æ–≥–¥–∞ –∏ —Ç–æ–ª—å–∫–æ —Ç–æ–≥–¥–∞, –∫–æ–≥–¥–∞ $B = A$ \gray{[–∏–∑ —Ä–∞–≤–µ–Ω—Å—Ç–≤–∞: $A ‚äÜ B$, –∏–∑ –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è: $|B| ‚â§ |A|$]}.
–û—Ç—Å—é–¥–∞ $\sum_{z‚àà(V_A + b)} y_z = u_A$ –¥–ª—è –≤—Å–µ—Ö $2^{m-r}$ —Å–º–µ–∂–Ω—ã—Ö –∫–ª–∞—Å—Å–æ–≤ –≤–∏–¥–∞ $V_A + b$ –µ—Å–ª–∏ $y = \Eval(f)$.
–ü–æ—Å–∫–æ–ª—å–∫—É –º—ã –¥–æ–ø—É—Å—Ç–∏–ª–∏, —á—Ç–æ $y$ –∏ $\Eval(f)$ –æ—Ç–ª–∏—á–∞—é—Ç—Å—è –Ω–µ –±–æ–ª–µ–µ —á–µ–º –≤ $2^{m - r - 1}$ –ø–æ–∑–∏—Ü–∏—è—Ö, –µ—Å—Ç—å –º–µ–Ω—å—à–µ —á–µ–º $2^{m - r - 1}$ —Å–º–µ–∂–Ω—ã—Ö –∫–ª–∞—Å—Å–æ–≤, –≤ –∫–æ—Ç–æ—Ä—ã—Ö $\sum_{z‚àà(V_A + b)} y_z ‚â† u_A$. –ü–æ—Å–ª–µ –≥–æ–ª–æ—Å–æ–≤–∞–Ω–∏—è –±–æ–ª—å—à–∏–Ω—Å—Ç–≤–æ–º* —Å—Ä–µ–¥–∏ —ç—Ç–∏—Ö $2^{m-r}$ —Å—É–º–º, –º—ã –Ω–∞–π–¥—ë–º –ø—Ä–∞–≤–∏–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ $u_A$.

–ü–æ—Å–ª–µ –≤—ã—á–∏—Å–ª–µ–Ω–∏—è –≤—Å–µ—Ö –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç–æ–≤ –ø—Ä–∏ –º–æ–Ω–æ–º–∞—Ö —Å—Ç–µ–ø–µ–Ω–∏ $r$, –º—ã –º–æ–∂–µ–º –ø–æ—Å—á–∏—Ç–∞—Ç—å:
\[
    y' = y - \Eval\left(\sum_{\substack{B‚äÜ\{1,‚Ä¶,m\}\\|B|=r}} u_B x_B\right).
\]

–≠—Ç–æ –∑–∞—à—É–º–ª–µ–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è –∫–æ–¥–æ–≤–æ–≥–æ —Å–ª–æ–≤–∞ $\Eval(f - \sum_{B‚äÜ\{1,‚Ä¶,m\},|B|=r} u_B x_B) ‚àà \RM(r-1, m)$, –∏ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –æ–æ—à–∏–±–æ–∫ –≤ $y'$ –º–µ–Ω—å—à–µ —á–µ–º $2^{m - r - 1}$ –∏–∑ –ø—Ä–µ–¥–ø–æ–ª–æ–∂–µ–Ω–∏—è. –¢–æ–≥–¥–∞ –º—ã –º–æ–∂–µ–º –∞–Ω–∞–ª–æ–≥–∏—á–Ω—ã–º –æ–±—Ä–∞–∑–æ–º –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–∏—Ç—å –≤—Å–µ –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã –ø—Ä–∏ –º–æ–Ω–æ–º–∞—Ö —Å—Ç–µ–ø–µ–Ω–∏ $r-1$ –∏—Å–ø–æ–ª—å–∑—É—è $y'$. –ü–æ–≤—Ç–æ—Ä—è—Ç—å —ç—Ç—É –ø—Ä–æ—Ü–µ–¥—É—Ä—É –ø–æ–∫–∞ –Ω–µ –±—É–¥—É—Ç –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω—ã –≤—Å–µ –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã $f$.

\begin{theorem*}
    –ü—Ä–∏ –¥–µ–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–∏–∏ –∫–æ–¥–∞ $\RM(r, m)$ –¥–ª—è —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ $r$ –∏ —Ä–∞—Å—Ç—É—â–µ–≥–æ $m$, –∞–ª–≥–æ—Ä–∏—Ç–º –†–∏–¥–∞ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ —É—Å—Ç—Ä–∞–Ω—è–µ—Ç –ª—é–±—É—é –æ—à–∏–±–∫—É —Å –≤–µ—Å–æ–º –•—ç–º–º–∏–Ω–≥–∞ –Ω–µ –±–æ–ª—å—à–µ $2^{m - r - 1}$ –∑–∞ $O(n \log^r n)$ –ø–æ –≤—Ä–µ–º–µ–Ω–∏.
\end{theorem*}

\gray{[–≤ –∏—Å—Ç–æ—á–Ω–∏–∫–µ –æ–Ω–∞ –±–µ–∑ –¥–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–∞]}

\begin{algorithm}[H]
    \DontPrintSemicolon
    \caption{Reed's algorithm for decoding $\RM(r, m)$}
    \KwData{Parameters $r$ and $m$ of the RM code, and a binary vector $y = (y_z \mid z ‚àà F_2^m)$ of length $n = 2^m$}
    \KwResult{A codeword $c ‚àà \RM(r, m)$}

    $t \gets r$\;
    \While{$ t ‚â• 0 $}{
        \ForEach{subset $A ‚äÜ \{1, ‚Ä¶, m\}$ with $|A| = t$}{
            Calculate $\sum_{z ‚àà (V_A + b)} y_z$ for all the $2^{m-t}$ cosets of $V_A$\;
            $num1 \gets $ number of cosets $(V_A + b)$ such that $\sum_{z‚àà(V_A + b)} y_z = 1$\;
            $u_A \gets \mathbf{1}[num1 ‚â• 2^{m - t - 1}]$\;
        }
        $y \gets y - \Eval(\sum_{A ‚äÜ \{1,‚Ä¶,m\}, |A| = t}) u_A x_A$\;
        $t \gets t - 1$\;
    }
    $c \gets \Eval(\sum_{A ‚äÜ \{1,‚Ä¶,m\}, |A| ‚â§ r}) u_Ax_A$\;
    \KwRet{$c$}\;
\end{algorithm}
